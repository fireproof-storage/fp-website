Light up your data.

import Fireproof from '@fireproof-storage/fireproof';

const database = Fireproof.storage("my-app-db");
const ok = await database.put({
    name: 'alice',
    age: 42
});

const doc = await database.get(ok.id);

console.log(doc.name, doc.age); // 'alice', 42
